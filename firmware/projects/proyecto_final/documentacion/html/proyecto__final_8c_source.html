115void read_gesturesTask(void *pvParameter){
116 while(true){
117 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
118 int8_t gesture = GesturetrackpadRead();
119 switch (gesture) {
120 case GESTURE_UP:
121 if (brillo < 215) {
122 printf("brillo: %d \n", brillo);
123 brillo += 40;
124 printf("brillo: %d \n",brillo);
125 if (brillo > 255) brillo = 255;
126 }
127 if (neopixel_encendido) {
128 NeoPixelAllColor(NeoPixelRgb2Color(
129 (red * brillo) / 255,
130 (green * brillo) / 255,
131 (blue * brillo) / 255));
132 }
133 vTaskDelay(500 / portTICK_PERIOD_MS);
134 break;
135 case GESTURE_DOWN:
136 if (brillo > 40){
137 printf("brillo: %d \n",brillo);
138 brillo -= 40;
139 printf("brillo: %d \n",brillo);
140 NeoPixelAllColor(NeoPixelRgb2Color((red * brillo) / 255,(green * brillo) / 255,(blue * brillo) / 255));
141 }
142 if (brillo == 0){
143 break;
144 }
145 if (neopixel_encendido){
146 NeoPixelAllColor(NeoPixelRgb2Color((red * brillo) / 255,(green * brillo) / 255,(blue * brillo) / 255));
147 }
148 vTaskDelay(500 / portTICK_PERIOD_MS);
149 break;
150 case GESTURE_FORWARD:
151 neopixel_encendido = true;
152 NeoPixelAllColor(NeoPixelRgb2Color(red, green, blue)); //enciende todos los leds de blanco con el brillo al maximo
153 break;
154 case GESTURE_BACK:
155 neopixel_encendido = false;
156 NeoPixelAllOff(); //apago la tira led
157 break;
158 default:
159 break;
160 }
161 }
162}

208void app_main(void){
209 static neopixel_color_t color[12];
210
211 timer_config_t timer_config_a = {
212 .timer = TIMER_A,
213 .period = SAMPLE_PERIOD_US,
214 .func_p = FuncTimerA,
215 .param_p = NULL
216 };
217 TimerInit(&timer_config_a);
218
219 ble_config_t ble_configuration = {
220 "ESP_EDU_MVB",
221 read_data // funcion de callback para cambiar el color desde el bluetooth
222 };
223
224 LedsInit();
225 BleInit(&ble_configuration);
226 NeoPixelInit(BUILT_IN_RGB_LED_PIN, LED_LENGTH, color);
227 GesturetrackpadInit();
228 xTaskCreate(&read_gesturesTask, "Read gestures", 2048, NULL, 5, &read_gestures_task_handle);
229 TimerStart(TIMER_A);
230
231 while(1){
232 vTaskDelay(CONFIG_BLINK_PERIOD / portTICK_PERIOD_MS);
233 switch(BleStatus()){ //dependiendo del estado del bluetooth se apaga, prende o togglea el led
234 case BLE_OFF:
235 LedOff(LED_BT);
236 break;
237 case BLE_DISCONNECTED:
238 LedToggle(LED_BT);
239 break;
240 case BLE_CONNECTED:
241 LedOn(LED_BT);
242 break;
243 }
244 }
245}
